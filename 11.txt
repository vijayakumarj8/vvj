11. Write a program to perform logical operations using Embedded C. 


#include <reg51.h>

void main(void) {
    unsigned char num1 = 0x55; // 01010101 in binary
    unsigned char num2 = 0xAA; // 10101010 in binary
    unsigned char and_result, or_result, xor_result, not_result;
    
    
        and_result = num1 & num2;
        or_result = num1 | num2;
        xor_result = num1 ^ num2;
        not_result = ~num1;
		
		P0 = and_result;
		P1 = or_result;
		P2 = xor_result;
		P3 = not_result;
        
        
}

output:

P0: Outputs 0x00 (binary 00000000).
P1: Outputs 0xFF (binary 11111111).
P2: Outputs 0xFF (binary 11111111).
P3: Outputs 0xAA (binary 10101010).
